Queue implementation with front and rear

#include <stdio.h>
#define size 10
int rear = -1;
int arr[size];

void enqueue(int val) {  // Enqueue only check based on rear
    if (rear == size - 1) {
        printf("\nQuery is full");
    }
    arr[++rear] = val;
}

int dequeue() {  // Dequeue only check based on front
    if (rear == -1) {
        return -1;
    }
    int temp = arr[0];
    for (int i = 1; i < size; i++) {
        arr[i - 1] = arr[i];
    }
    rear--;
    return temp;
}

void main() {
    for (int i = 1; i <= size; i++) {
        printf("Enqueued: %d\n", i);
        enqueue(i);
    }
    printf("/**************/\n");
    for (int i = 0; i < 12; i++) {
        int temp = dequeue();
        if (temp == -1) {
            printf("Queue is empty\n");
        } else {
            printf("Dequeued: %d\n", temp);
        }
    }
}

Normal Queue only using rear

#include <stdio.h>
#define size 10

int front = -1;
int rear;
int arr[size];

void enqueue(int val) {  // Enqueue only check based on rear
    if (front == -1) {
        arr[0] = val;
        front++;
        return;
    }
    if (rear == size - 1) {
        printf("\nQuery is full");
    }
    arr[++rear] = val;
}

int dequeue() {  // Dequeue only check based on front
    if (front == rear) {
        printf("\nQueue is empty");
    } else {
        int temp = arr[front];
        front++;
        return temp;
    }
    return -1;
}

void main() {
    for (int i = 1; i <= size; i++) {
        printf("\nEnqueued: %d", i);
        enqueue(i);
    }
    for (int i = 0; i < size; i++) {
        int temp = dequeue();
        printf("\nDequeued: %d", temp);
    }
}

Circular Queue

Normal circular Queue:

#include <stdio.h>
#define SIZE 5
int front = -1;
int rear = -1;
int items[SIZE];

int isFull() {
    if (front == 0 && rear == SIZE - 1) {
        return 1;
    }
    if (front == rear + 1) {
        return 1;
    }
    return 0;
}

int isEmpty() {
    if (front == -1)
        return 1;
    else
        return 0;
}

void enQueue(int element) {
    if (isFull()) {
        printf("Queue is full");
    } else {
        if (front == -1) {
            front = 0;
        }
        rear = (rear + 1) % SIZE;
        items[rear] = element;
        printf("Enqueued: %d\n", element);
    }
}

void deQueue() {
    int element;
    if (isEmpty()) {
        printf("Queue is empty");
    } else {
        element = items[front];
        if (front == rear) {
            front = -1;
            rear = -1;
        } else {
            front = (front + 1) % SIZE;
        }
        printf("Dequeued: %d\n", element);
    }
}

void main() {
    for (int i = 0; i < SIZE; i++) {
        enQueue(i);
    }
    for (int i = 0; i < SIZE; i++) {
        deQueue();
    }
    deQueue();
}

Circular Queue using Struct

#include <stdio.h>
#define SIZE 5

struct Cir_Queue {
    int front;
    int rear;
    int can[SIZE];
};

int isFull(struct Cir_Queue* queue) {
    if (queue->front == 0 && queue->rear == SIZE - 1) {
        return 1;
    }
    if (queue->front == queue->rear + 1) {
        return 1;
    }
    return 0;
}

int isEmpty(struct Cir_Queue* queue) {
    if (queue->front == -1)
        return 1;
    else
        return 0;
}

void enQueue(struct Cir_Queue* queue, int element) {
    if (isFull(queue)) {
        printf("Queue is full");
    } else {
        if (queue->front == -1) {
            queue->front = 0;
        }
        queue->rear = (queue->rear + 1) % SIZE;
        queue->can[queue->rear] = element;
        printf("Enqueued: %d\n", element);
    }
}

void deQueue(struct Cir_Queue* queue) {
    int element;
    if (isEmpty(queue)) {
        printf("Queue is empty");
    } else {
        element = queue->can[queue->front];
        if (queue->front == queue->rear) {
            queue->front = -1;
            queue->rear = -1;
        } else {
            queue->front = (queue->front + 1) % SIZE;
        }
        printf("Dequeued: %d\n", element);
    }
}

int main() {
    struct Cir_Queue c1;
    c1.front = -1;
    c1.rear = -1;

    for (int i = 0; i < SIZE; i++) {
        enQueue(&c1, i);
    }
    for (int i = 0; i < SIZE; i++) {
        deQueue(&c1);
    }
    deQueue(&c1);

    return 0;
}
